//关键字 + 变数名称：类型 = 值

print(number)
print("Hello World")

 number =2

//实作：打招呼

var name = "Steve"
let location = "China"
print("Hello", name,"from", location )
print("Hello \(name) from \(location) )

name = "Jane"
print("Hello \(name) from \(location) )
